  9  * @param {string} [currency] The currency ISO code for this value.
 10  * @returns {fMoney} The money object.
 11  */
 12 var fMoney = function (amount, currency) {
 13   if (fMoney.defaultCurrency === 'INVALID') {
 14     fCore.debug('No default currency has been set.');
 15     return this;
 16   }
 17 
 18   /**
 19    * @private
 20    * @type string|null
 21    */
 22   this.currency = (currency !== undefined) ? currency : fMoney.defaultCurrency;
 23 
 24   /**
 25    * @private
 26    * @type number
 27    */
 28   var precision =  +fMoney.getCurrencyInfo(this.currency, fMoney.CURRENCY_INFO_ELEMENT_PRECISION);
 29 
 30   /**
 31    * @private
 32    * @type fNumber
 33    */
 34   this.amount = new fNumber(amount, precision);
 35 
 36   return this;
 37 };

 73 /**
 74  * Define a new currency for fMoney
 75  * @param {string} isoCode The ISO code for the currency.
 76  * @param {string} name The name of the currency
 77  * @param {string} symbol The symbol associated with the currency
 78  * @param {number} precision The number of digits after the decimal
 79  * @param {string} value The value of the currency
 80  * @see http://en.wikipedia.org/wiki/ISO_4217
 81  */
 82 fMoney.defineCurrency = function (isoCode, name, symbol, precision, value) {
 83   fMoney._currencies[isoCode] = {};
 84   fMoney._currencies[isoCode][fMoney.CURRENCY_INFO_ELEMENT_NAME] = name;
 85   fMoney._currencies[isoCode][fMoney.CURRENCY_INFO_ELEMENT_SYMBOL] = symbol;
 86   fMoney._currencies[isoCode][fMoney.CURRENCY_INFO_ELEMENT_PRECISION] = precision;
 87   fMoney._currencies[isoCode][fMoney.CURRENCY_INFO_ELEMENT_VALUE] = value;
 88 };
 89 /**
 90  * List all the defined currencies.
 91  * @param {boolean} [alpha=false] Set to <em>true</em> if you want the
 92  *   currencies in alphabetical order.
 93  * @returns {Array} The currency ISO codes. If you set alpha to true they will be
 94  * alphabetical order if not they will be in a unsorted order
 95  */
 96 fMoney.getCurrencies = function (alpha) {
 97   if (alpha === undefined) {
 98     alpha = false;
 99   }
100   return objectKeys(fMoney._currencies, alpha);
101 };
102 /**
103  * Retrieve information about a defined currency
104  * @param {string} isoCode The iso code of the currency you want to retrieve
105  * @param {number} [element] The element of the currency you want to retrieve.
106  *   One of the fMoney.CURRENCY_INFO_* consts.
107  * @returns {string|Object|number} The information about the retrieved currency.
108  */
109 fMoney.getCurrencyInfo = function (isoCode, element) {
110   if (fMoney._currencies[isoCode] === undefined) {
111     fCore.debug('The currency specified, %s, is not a valid currency. Returning empty object.', isoCode);
112     return {};
113   }
114 
115   if (element === undefined) {
116     return fMoney._currencies[isoCode];
117   }
118 
119   return fMoney._currencies[isoCode][element];
120 };

226 /**
227  * Turns a string into an fMoney object if a default currency is defined.
228  * @private
229  * @param {fMoney|string|number} money The value to convert to an fMoney
230  *   object.
231  * @returns {fMoney|number|string|null} The converted value; null if the value
232  *   was invalid or if the default currency is not set for the class.
233  */
234 fMoney.prototype.makeMoney = function (money) {
235   // All fMoney objects have a makeMoney property
236   if (money.makeMoney) {
237     return money;
238   }
239 
240   // All JS objects except null and undefined have a toString() method
241   if (money && money.toString) {
242     money = money.toString();
243   }
244 
245   // Now just make sure it's not empty, or somehow boolean false
246   if (!money) {
247     fCore.debug('The money value specified is not valid.');
248     return null;
249   }
250 
251   // Class validation
252   if (fMoney.defaultCurrency === 'INVALID') {
253     fCore.debug('A default currency must be set in order to convert strings or numbers to fMoney objects on the fly.');
254     return null;
255   }
256 
257   return new fMoney(money);
258 };
259 /**
260  * Formats the amount by preceeding the amount with the currency symbol
261  *   and adding thousand seperators.
262  * @param {boolean} removeZeroFraction If true and all the digits after the
263  *   decimal place are 0's the decimal place and all zeros are removed.
264  * @returns {string} The formatted (and possibly converted) value.
265  */
266 fMoney.prototype.format = function (removeZeroFraction) {
267   var number = this.toString();
268   var parts = number.split('.');
269 
270   var integer = parts[0];
271   var fraction = (parts[1] === undefined) ? '' : parts[1];
272 
273   var sign = '';
274   if (integer.charAt(0) === '-'){
275     sign = '-';
276     integer = integer.substr(1);
277   }
278 
279   var intSections = [];
280 
281   for (var i = integer.length-3; i > 0; i -= 3) {
282     intSections.unshift(integer.substr(i, 3));
283   }
284   intSections.unshift(integer.substr(i, 3));
285 
286   var symbol = fMoney.getCurrencyInfo(this.currency, fMoney.CURRENCY_INFO_ELEMENT_SYMBOL);
287   integer = intSections.join(',');
288   fraction = (fraction.length) ? '.' + fraction : '';
289 
290   if (removeZeroFraction && fUTF8.rtrim(fraction, '.0') === '') {
291     fraction = '';
292   }
293 
294   return sign + symbol + integer + fraction;
295 };
