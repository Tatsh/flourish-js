  7 var fDate = function (date) {
  8   /**
  9    * @type number|boolean
 10    * @private
 11    */
 12   var timestamp;
 13 
 14   /**
 15    * @type number
 16    * @private
 17    */
 18   this._date = 0;
 19 
 20   if (date === undefined || date === 'CURRENT_TIMESTAMP' || date === 'CURRENT_DATE') {
 21     timestamp = fDate._now();
 22   }
 23   else if (typeof date === 'number' && date.toString().match(/^-?\d+$/)) {
 24     timestamp = date;
 25   }
 26   else if (date.lt) {
 27     timestamp = date._date;
 28   }
 29   else {
 30     if (date.toString) {
 31       date = date.toString();
 32     }
 33 
 34     // TODO
 35     //date = fTimestamp.callUnformatCallback(date);
 36     //date = fTimestamp.fixISOWeek(date);
 37     timestamp = parseInt(strtotime(date), 10);
 38   }
 39 
 40   if (timestamp === false) {
 41     fCore.debug('The date specified, %s,  does not appear to be a valid date.', date && date.toString ? date.toString() : '');
 42     return this;
 43   }
 44 
 45   if (timestamp < 9999999999) {
 46     timestamp *= 1000;
 47   }
 48 
 49   date = new Date(timestamp);
 50   // Y-m-d 00:00:00
 51   var test = parseInt(strtotime(date.getFullYear() + ' ' + (date.getMonth()+1) + ' ' + date.getDate() + ' 00:00:00', timestamp), 10);
 52   if (!isNaN(test)) {
 53     this._date = test;
 54   }
 55 };
 56 /**
 57  * @const
 58  * @type Object
 59  * @private
 60  */
 61 fDate._breakPoints =  {
 62   432000:     [86400, 'day', 'days'],
 63   1814400:    [604800, 'week', 'weeks'],
 64   23328000:   [2592000, 'month', 'months'],
 65   2147483647: [31536000, 'year', 'years']
 66 };
 67 /**
 68  * @returns {number}
 69  * @private
 70  */
 71 fDate._now = function () {
 72   return parseInt((new Date()).getTime() / 1000, 10);
 73 };
 74 /**
 75  * Composes text.
 76  * @param {string} message The text string.
 77  * @returns {string} String.
 78  * @private
 79  */
 80 fDate.compose = function (message) {
 81   return sprintf.apply(fDate, arguments);
 82 };
 83 /**
 84  * Adjusts the date with the specified adjustment.
 85  * @param {string} adjustment The adjustment to make.
 86  * @returns {fDate} The new date if updates are needed.
 87  */
 88 fDate.prototype.adjust = function (adjustment) {
 89   var timestamp = strtotime(adjustment, this._date);
 90 
 91   if (timestamp === false || timestamp === -1) {
 92     fCore.debug('The adjustment specified, %s, does not appear to be a valid relative date measurement.', adjustment);
 93     return new fDate(fDate._now());
 94   }
 95 
 96   return new fDate(timestamp);
 97 };
 98 /**
 99  * Checks if the date is equal to another date.
100  * @param {fDate|Object|string|number} [otherDate] The date to compare with.
101  * @returns {boolean} If the dates are equal.
102  */
103 fDate.prototype.eq = function (otherDate) {
104   otherDate = new fDate(otherDate);
105   return this._date === otherDate._date; // The PHP does it this way
106 };
107 /**
108  * Get a fuzzy difference string. Unlike Flourish, this only compares against
109  *   'now'.
110  * @return {string} A fuzzy time string in English.
111  */
112 fDate.prototype.getFuzzyDifference = function () {
113   // TODO Complete implementation
114   return (function (breakPoints, value) {
115     var now = fDate._now();
116     var diff = now - value;
117     var unitDiff = 0, units = 'days';
118 
119     if (Math.abs(diff) < 10) {
120       return 'right now';
121     }
122 
123     for (var point in breakPoints) {
124       if (breakPoints.hasOwnProperty(point)) {
125         if (Math.abs(diff) > point) {
126           continue;
127         }
128 
129         unitDiff = Math.round(Math.abs(diff) / breakPoints[point][0]);
130         units = fGrammar.inflectOnQuantity(unitDiff, breakPoints[point][1], breakPoints[point][2]);
131         break;
132       }
133     }
134 
135     if (diff < 0) {
136       return unitDiff + ' ' + units + ' from now';
137     }
138 
139     return unitDiff + ' ' + units + ' ago';
140   })(fDate._breakPoints, this._date);
141 };

178 /**
179  * Implements toString().
180  * @returns {string} The date in YYYY-mm-dd format.
181  */
182 fDate.prototype.toString = function () {
183   var date = new Date(this._date * 1000);
184   var month = date.getMonth()+1;
185   if (month < 10) {
186     month = '0' + month;
187   }
188   var day = date.getDate();
189   if (day < 10) {
190     day = '0' + day;
191   }
192   return date.getFullYear() + '-' + month + '-' + day;
193 };
