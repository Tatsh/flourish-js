1 /**
  2  * Provides request-related methods.
  3  * @constructor
  4  */
  5 var fRequest = function () {};
  6 /**
  7  * @type (Object|null)
  8  * @private
  9  */
 10 fRequest._REQUEST = null;
 11 
 12 /**
 13  * Casts the value to a type specified.
 14  * @private
 15  * @param {string} value Value to cast.
 16  * @param {string} to Cast to type. One of 'number' or 'float'.
 17  * @returns {number|string|boolean}
 18  */
 19 fRequest._cast = function (value, to) {
 20   var lcValue, trues = {
 21     'true': 1,
 22     'yes': 1,
 23     'on': 1
 24   };
 25 
 26   switch (to) {
 27     case 'number':
 28       return parseInt(value, 10);
 29 
 30     case 'float':
 31       return parseFloat(value);
 32 
 33     case 'bool':
 34     case 'boolean':
 35       lcValue = value.toLowerCase();
 36       return trues[lcValue] !== undefined;
 37   }
 38 
 39   return value;
 40 };
 41 /**
 42  * Initialises the class.
 43  * @private
 44  */
 45 fRequest._initialize = function () {
 46   if (fRequest._REQUEST === null) {
 47     var qs = fURL.getQueryString();
 48     var items = qs.split('&');
 49     var itemsAsObject = {}, current;
 50 
 51     for (var i = 0; i < items.length; i++) {
 52       current = items[i].split('=');
 53       itemsAsObject[current[0]] = current[1];
 54     }
 55 
 56     fRequest._REQUEST = itemsAsObject;
 57   }
 58 };
 59 /**
 60  * Gets a value from the query string.
 61  * @param {string} name The key to get the value of.
 62  * @param {string} [castTo=null] Cast the value to this data type. One of:
 63  *   'number', 'float', 'bool', 'boolean'.
 64  * @param {string|number|null|boolean} [defaultValue=null] The default value to
 65  *   return if the key does not exist.
 66  * @param {boolean} [useDefaultForBlank=false] If the request value is a blank
 67  *   string and <code>defaultValue</code> is specified,
 68  *   <code>defaultValue</code> will be returned.
 69  * @returns {string|null|number|boolean} The value or null if the value does
 70  *   not exist and <code>defaultValue</code> is not specified.
 71  */
 72 fRequest.get = function (name, castTo, defaultValue, useDefaultForBlank) {
 73   fRequest._initialize();
 74 
 75   useDefaultForBlank === undefined && (useDefaultForBlank = false);
 76 
 77   if (defaultValue === undefined) {
 78     defaultValue = null;
 79   }
 80   if (castTo === undefined) {
 81     castTo = 'string';
 82   }
 83 
 84   if (!fURL.getQueryString() || fRequest._REQUEST[name] === undefined ||
 85     (fRequest._REQUEST[name] === '' && useDefaultForBlank)) {
 86     return defaultValue;
 87   }
 88 
 89   var ret = fRequest._cast(fRequest._REQUEST[name], castTo);
 90   if (isNaN(ret) && (castTo === 'number' || castTo === 'float')) {
 91     ret = null;
 92   }
 93 
 94   return ret;
 95 };
