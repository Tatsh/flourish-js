  3 /**
  4  * Provides low-level debugging functionality.
  5  * @constructor
  6  */
  7 var fCore = function () {};
  8 /**
  9   * @type boolean
 10   * @private
 11   */
 12 fCore._debugEnabled = false;
 13 /**
 14   * Returns is the current OS is one of the OSes passed as a parameter.
 15   *
 16   * Valid OS strings are:
 17   * <ul>
 18   * <li>'linux'</li>
 19   * <li>'bsd'</li>
 20   * <li>'osx'</li>
 21   * <li>'windows'</li>
 22   * </ul>
 23   *
 24   * Others may work.
 25   * @param [args] Variable number of arguments.
 26   * @returns {boolean} If the current OS is included in the list of OSes passed
 27   *   as parameters.
 28   */
 29 fCore.checkOS = function (args) {
 30   var os = [], i;
 31   var ua = navigator.userAgent;
 32 
 33   for (i = 0; i < arguments.length; i++) {
 34     if (!arguments[i]) {
 35       continue;
 36     }
 37 
 38     if (ua.match(new RegExp(arguments[i], 'i'))) {
 39       return true;
 40     }
 41   }
 42 
 43   return false;
 44 };
 45 /**
 46   * Logs a message.<p>Must call <code>fCore.enableDebugging()</code> first
 47   *   with <code>true</code> or this method does nothing.</p><p>To see messages
 48   *   in IE versions prior to 8, you must have debugging enabled ('Disable
 49   *   Debugging' not checked) and have Visual Studio attached to the Script
 50   *   process (<em>Script,x86</em>).</p>
 51   * @param {string} fmt The format string is composed of zero or more
 52   *   directives: ordinary characters (excluding <code>%</code>) that are
 53   *   copied directly to the result, and conversion specifications, each of
 54   *   which results in fetching its own parameter.
 55   * @param {string|number} [args] Variable number of arguments specified
 56   *   according to the format string.
 57   */
 58 fCore.debug = function (fmt, args) {
 59   if (fCore._debugEnabled) {
 60     var message = sprintf.apply(fCore, arguments);
 61 
 62     if (window['Debug'] && window['Debug'].writeln) {
 63       window['Debug'].writeln(message);
 64     }
 65     else if (window['opera'] && window['opera'].postError) {
 66       window['opera'].postError(message);
 67     }
 68     else if (window.console) {
 69       window.console.log(message);
 70     }
 71   }
 72 };
 73 /**
 74   * Enables or disables debugging globally.
 75   * @param {boolean} bool Boolean value, true or false.
 76   * @see fCore#debug
 77   */
 78 fCore.enableDebugging = function (bool) {
 79   fCore._debugEnabled = bool ? true : false;
 80 };
 81 /**
 82   * If debugging is enabled.
 83   * @returns {boolean} If debugging is enabled.
 84   */
 85 fCore.getDebug = function () {
 86   return fCore._debugEnabled;
 87 };
