  3 /**
  4  * Provides low-level debugging functionality.
  5  * @constructor
  6  */
  7 var fCore = function () {};
  8 /**
  9   * @type boolean
 10   * @private
 11   */
 12 fCore._debugEnabled = false;
 13 /**
 14   * Returns is the current OS is one of the OSes passed as a parameter.
 15   *
 16   * Valid OS strings are:
 17   * <ul>
 18   * <li>'linux'</li>
 19   * <li>'bsd'</li>
 20   * <li>'osx'</li>
 21   * <li>'windows'</li>
 22   * </ul>
 23   *
 24   * Others may work.
 25   * @param [args] Variable number of arguments.
 26   * @returns {boolean} If the current OS is included in the list of OSes passed
 27   *   as parameters.
 28   */
 29 fCore.checkOS = function (args) {
 30   var os = [], i;
 31   var ua = navigator.userAgent;
 32 
 33   for (i = 0; i < arguments.length; i++) {
 34     if (!arguments[i]) {
 35       continue;
 36     }
 37 
 38     if (ua.match(new RegExp(arguments[i], 'i'))) {
 39       return true;
 40     }
 41   }
 42 
 43   return false;
 44 };
 45 /**
 46   * Logs a message.<p>Must call <code>fCore.enableDebugging()</code> first
 47   *   with <code>true</code> or this method does nothing.</p><p>To see messages
 48   *   in IE versions prior to 8, you must have debugging enabled ('Disable
 49   *   Debugging' not checked) and have Visual Studio attached to the Script
 50   *   process (<em>Script,x86</em>).</p>
 51   * <p>If the arguments passed are valid for sprintf() it will be used.
 52   *   Otherwise, the <code>arguments</code> will be logged.</p>
 53   * @param {*} fmt The format string is composed of zero or more directives:
 54   *   ordinary characters (excluding <code>%</code>) that are copied directly
 55   *   to the result, and conversion specifications, each of which results in
 56   *   fetching its own parameter.
 57   * @param {string|number} [args] Variable number of arguments specified
 58   *   according to the format string.
 59   */
 60 fCore.debug = function (fmt, args) {
 61   if (fCore._debugEnabled) {
 62     var validForSprintf = typeof arguments[0] === 'string';
 63 
 64     // Only strings, numbers, and arrays are allowed for sprintf past the first argument
 65     if (validForSprintf) {
 66       for (var i = 1; i < arguments.length; i++) {
 67         if (objectIsObject(arguments[i])) {
 68           validForSprintf = false;
 69           break;
 70         }
 71 
 72         if (typeof arguments[i] !== 'number' && typeof arguments[i] !== 'string' && !arrayIsArray(arguments[i])) {
 73           validForSprintf = false;
 74           break;
 75         }
 76       }
 77     }
 78 
 79     // If all of the arguments are compatible with sprintf, call it
 80     // IE8 and 9 cannot use .apply() with console.log
 81     if (window.console && window.console.log) {
 82       var callNormal = function () {
 83         window.console.log(validForSprintf ? sprintf.apply(fCore, arguments) : arguments);
 84       };
 85       if (window.console.log.apply) {
 86         try {
 87           window.console.log.apply(window.console, validForSprintf ? [sprintf.apply(fCore, arguments)] : arguments);
 88         }
 89         catch (e) {
 90           callNormal();
 91         }
 92       }
 93       else {
 94         callNormal();
 95       }
 96     }
 97     else if (window['Debug'] && window['Debug'].writeln) {
 98       window['Debug'].writeln.apply(window, validForSprintf ? [sprintf.apply(fCore, arguments)] : arguments);
 99     }
100     else if (window['opera'] && window['opera'].postError) {
101       window['opera'].postError(validForSprintf ? sprintf.apply(fCore, arguments) : arguments);
102     }
103   }
104 };
105 /**
106   * Enables or disables debugging globally.
107   * @param {boolean} bool Boolean value, true or false.
108   * @see fCore#debug
109   */
110 fCore.enableDebugging = function (bool) {
111   fCore._debugEnabled = bool ? true : false;
112 };
113 /**
114   * If debugging is enabled.
115   * @returns {boolean} If debugging is enabled.
116   */
117 fCore.getDebug = function () {
118   return fCore._debugEnabled;
119 };
