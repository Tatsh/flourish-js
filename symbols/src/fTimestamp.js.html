1 /**
  2  * Represents a date and time as a value object.
  3  * @constructor
  4  * @param {fTimestamp|Object|string|number|Date|null} [datetime=null] The
  5  *   date/time to represent. <code>null</code> is interpreted as now.
  6  * @param {string|null} [timezone=null] The timezone for the date/time. This
  7  *   causes date/time to be interpreted as being in the specified timezone.
  8  * @return {fTimestamp} The timestamp object.
  9  */
 10 var fTimestamp = function (datetime, timezone) {
 11   /**
 12    * @type string
 13    * @private
 14    */
 15   var defaultTimezone = fTimestamp.getDefaultTimezone();
 16   /**
 17    * @type number
 18    * @private
 19    */
 20   var numberDatetime = parseInt(datetime, 10);
 21   /**
 22    * @type number
 23    * @private
 24    */
 25   var timestamp;
 26 
 27   if (timezone !== undefined) {
 28     if (!fTimestamp.isValidTimezone(timezone.toString())) {
 29       fCore.debug('The timezone specified, %s, is not a valid timezone.', timezone);
 30       return null;
 31     }
 32   }
 33   else if (datetime && datetime.timezone) {
 34     timezone = datetime.timezone;
 35   }
 36   else {
 37     timezone = defaultTimezone;
 38   }
 39 
 40   this.timezone = timezone;
 41 
 42   if (!datetime || datetime.toString().substr(0, 12).toUpperCase()  === 'CURRENT_TIME') {
 43     timestamp = (new Date()).getTime() / 1000;
 44   }
 45   else if (!isNaN(numberDatetime) && datetime.toString().match(/^\-?\d+$/)) {
 46     timestamp = numberDatetime;
 47   }
 48   else if (datetime.timestamp) {
 49     timestamp = datetime.timestamp;
 50   }
 51   else if (datetime.toString().toUpperCase() === 'CURRENT_DATE') {
 52     var date = new Date();
 53     var month = (date.getMonth() + 1).toString();
 54     var day = date.getDate().toString();
 55 
 56     if (day.length < 2) {
 57       day = '0' + day;
 58     }
 59     if (month.length < 2) {
 60       month = '0' + month;
 61     }
 62 
 63     timestamp = +strtotime(date.getFullYear() + '-' + month + '-' + day);
 64   }
 65   else {
 66     if (typeof datetime === 'object') {
 67       if (datetime.getTime) { // Handle if a Date object is passed
 68         timestamp = datetime.getTime() / 1000;
 69       }
 70 //       else if (datetime.toString || numberDatetime === datetime) {
 71 //         datetime = date.toString();
 72 //       }
 73 //
 74 //       // datetime = fTimestamp.callUnformatCallback(datetime);
 75 //
 76 //       if (timezone !== defaultTimezone) {
 77 //
 78 //       }
 79 //
 80 //       timestamp = strtotime(fTimestamp.fixISOWeek(datetime));
 81 //
 82 //       if (timezone !== defaultTimezone) {
 83 //
 84 //       }
 85     }
 86   }
 87 
 88   if (timestamp === false) {
 89     fCore.debug('The date/time specified, %s, does not appear to be a valid date/time', datetime.toString());
 90     return null;
 91   }
 92 
 93   /**
 94    * @type {number}
 95    */
 96   this.timestamp = parseInt(timestamp, 10);
 97 
 98   return this;
 99 };

618 fTimestamp.isValidTimezone = function (timezone) {

620 };
621 /**
622  * Set the default timezone. Does nothing if the <code>timezone</code>
623  *   parameter is invalid.
624  * @param {string} timezone A timezone string, such as America/Chicago.
625  */
626 fTimestamp.setDefaultTimezone = function (timezone) {
627   if (!fTimestamp.isValidTimezone(timezone)) {
628     return;

630   fTimestamp._defaultTimezone = timezone;

653 /**
654  * Returns the approximate difference in time, discarding any unit of measure
655  *   but the least specific.
656  * @param {fTimestamp|Date|string|number|null} [otherTimestamp] The timestamp
657  *   to compare with. If not specified or null, compares against 'now'.
658  * @param {boolean} [simple=false] When true, the returned value will include
659  *   the difference in the two timestamps, but not 'from now', 'ago', 'after', or 'before'.
660  * @return {string} The fuzzy time in English.
661  */
662 fTimestamp.prototype.getFuzzyDifference = function (otherTimestamp, simple) {
663   // Does not support having simple be the first argument
664 
665   var relativeToNow = false;
666   if (!otherTimestamp) {
667     relativeToNow = true;
668   }
669   otherTimestamp = new fTimestamp(otherTimestamp);
670 
671   return (function (breakPoints, value, otherValue) {
672     var diff = value.timestamp - otherTimestamp.timestamp;
673     var unitDiff = 0, units = 'seconds';
674 
675     if (Math.abs(diff) < 10) {
676       if (relativeToNow) {
677         return 'right now';
678       }
679       return 'at the same time';
680     }
681 
682     for (var point in breakPoints) {
683       if (breakPoints.hasOwnProperty(point)) {
684         if (Math.abs(diff) > point) {
685           continue;
686         }
687 
688         unitDiff = Math.round(Math.abs(diff) / breakPoints[point][0]);
689         units = fGrammar.inflectOnQuantity(unitDiff, breakPoints[point][1], breakPoints[point][2]);
690         break;
691       }
692     }
693 
694     var prefix = '%1$s %2$s';
695 
696     if (simple) {
697       return sprintf(prefix, unitDiff, units);
698     }
699 
700     if (relativeToNow) {
701       if (diff > 0) {
702         return sprintf(prefix + ' from now', unitDiff, units);
703       }
704       return sprintf(prefix + ' ago', unitDiff, units);
705     }
706 
707     if (diff > 0) {
708       return sprintf(prefix + ' after', unitDiff, units);
709     }
710 
711     return sprintf(prefix + ' before', unitDiff, units);
712   })(fTimestamp._breakPoints, this, otherTimestamp);
713 };
