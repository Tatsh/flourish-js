  2 /**
  3  * Front end to web localStorage object for browsers that support it. There is
  4  *   no fallback yet.
  5  * @constructor
  6  */
  7 var fSession = function () {};
  8 /**
  9  * If session storage is supported.
 10  * @type boolean
 11  * @private
 12  */
 13 fSession.isSupported = (function () {
 14   return 'sessionStorage' in window;
 15 })();
 16 /**
 17  * Local reference to session.
 18  * @type Storage|undefined
 19  * @private
 20  */
 21 fSession.store = window.sessionStorage;
 22 /**
 23  * Add a value. Object key access is not supported.
 24  * @param {string} key Key.
 25  * @param {string|number|boolean} value Value to store.
 26  * @param {boolean} beginning If the value should be added to the beginning.
 27  */
 28 fSession.add = function (key, value, beginning) {
 29   if (!fSession.isSupported) {
 30     return;
 31   }
 32 
 33   beginning === undefined && (beginning = false);
 34 
 35   var item = fSession.store.getItem(key);
 36 
 37   if (item === null) {
 38     item = '';
 39   }
 40 
 41   if (!beginning) {
 42     return fSession.store.setItem(key, item + String(value));
 43   }
 44 
 45   return fSession.store.setItem(key, String(value) + item);
 46 };
 47 /**
 48  * Removes all session values with the provided prefix.
 49  * @param {string|null} [prefix=null] Prefix to clear all session values for.
 50  */
 51 fSession.clear = function (prefix) {
 52   if (!fSession.isSupported) {
 53     return;
 54   }
 55 
 56   prefix === undefined && (prefix = null);
 57 
 58   if (!prefix) {
 59     return fSession.store.clear();
 60   }
 61 
 62   for (var key, prefixLength = prefix.length, i = 0; i < fSession.store.length; i++) {
 63     key = fSession.store.key(i);
 64     if (key.substr(0, prefixLength) === prefix) {
 65       fSession.store.removeItem(key);
 66     }
 67   }
 68 };
 69 /**
 70  * Delete a value. This cannot be called <code>delete</code> because
 71  *   <code>delete</code> is a reserved word in JavaScript.
 72  * @param {string} key Key. Object key access is not supported.
 73  * @param {string|null} [defaultValue=null] The default value to return if
 74  *   <code>key</code> is not set.
 75  * @returns {string|null} The value of the key that was deleted.
 76  */
 77 fSession.deleteKey = function (key, defaultValue) {
 78   if (defaultValue === undefined) {
 79     defaultValue = null;
 80   }
 81 
 82   if (!fSession.isSupported) {
 83     return defaultValue;
 84   }
 85 
 86   var returnValue = String(fSession.store.getItem(key));
 87 
 88   if (returnValue === 'null') {
 89     returnValue = defaultValue;
 90   }
 91 
 92   fSession.store.removeItem(key);
 93 
 94   return returnValue;
 95 };
 96 /**
 97  * Deletes all values in the session storage.
 98  */
 99 fSession.destroy = function () {
100   if (!fSession.isSupported) {
101     return;
102   }
103   fSession.store.clear();
104 };
105 /**
106  * Get data from the session storage.
107  * @param {string} key Key. Object key access is not supported.
108  * @param {string|null} [defaultValue=null] The default value to return if
109  *   <code>key</code> does not exist.
110  * @returns {string|null} The key value.
111  */
112 fSession.get = function (key, defaultValue) {
113   if (defaultValue === undefined) {
114     defaultValue = null;
115   }
116 
117   if (!fSession.isSupported) {
118     return defaultValue;
119   }
120 
121   var item = String(fSession.store.getItem(key));
122 
123   if (item === 'null') {
124     item = defaultValue;
125   }
126 
127   return item;
128 };
129 /**
130  * Set data to session storage.
131  * @param {string} key Key to set.
132  * @param {string|boolean|number} value Value to set.
133  */
134 fSession.set = function (key, value) {
135   if (!fSession.isSupported) {
136     return;
137   }
138 
139   value === undefined && (value = '');
140   return fSession.store.setItem(key, String(value));
141 };
