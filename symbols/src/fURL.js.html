1 /**
  2  * Provides functionality to retrieve and manipulate URL information.
  3  *
  4  * This classes uses <code>window.location</code> for all operations.
  5  * @constructor
  6  */
  7 var fURL = function () {};
  8 /**
  9  * Returns the current query string.
 10  * @returns {string} The query string.
 11  */
 12 fURL.getQueryString = function () {
 13   return location.search ? location.search.substr(1) : '';
 14 };
 15 /**
 16  * Returns the requested URL, does not include the domain name or query string.
 17  * @returns {string} The requested URL without the query string.
 18  */
 19 fURL.get = function () {
 20   return location.pathname;
 21 };
 22 /**
 23  * Returns the current domain name, with protocol prefix.
 24  * @returns {string} The current domain name, prefixed by <code>http://</code>
 25  *   or <code>https://</code>.
 26  */
 27 fURL.getDomain = function () {
 28   return location.protocol + '//' + location.hostname;
 29 };
 30 /**
 31  * Redirects to the URL specified, without requiring a fully-qualified URL.
 32  * @param {string} url The URL to redirect to.
 33  */
 34 fURL.redirect = function (url) {
 35   location.href = url;
 36 };
