  2 /**
  3  * Provides large/precise number support.<br><br>
  4  *
  5  * TODO<br>
  6  * Known issues:<br>
  7  * <ul>
  8  * <li>baseConvert does not work yet because it relies on .pow</li>
  9  * <li>.sub does not work when subracting the same number</li>
 10  * <li>.div causes 'Maximum call stack size exceeded'</li>
 11  * <li>.mul gives the incorrect result</li>
 12  * <li>.pow gives an incorrect result (relies on .mul working)</li>
 13  * </ul>
 14  *
 15  * @param {string|Object|number} value The number to use.
 16  * @param {number} [scale] The scale to use.
 17  * @constructor
 18  * @returns {fNumber} The number object.
 19  */
 20 var fNumber = function (value, scale) {
 21   if (scale === undefined) {
 22     scale = 0;
 23   }
 24 
 25   if (typeof value !== 'string') {
 26     value = String(value);
 27   }
 28 
 29   value = fNumber.parse(value, 'array');
 30 
 31   if (scale !== null) {
 32     if (value['fraction'].length > scale) {
 33       value['fraction'] = value['fraction'].substr(0, scale);
 34     }
 35     else {
 36       value['fraction'] = fUTF8.pad(value['fraction'], scale, '0', 'right');
 37     }
 38   }
 39 
 40   /**
 41    * @private
 42    * @type string
 43    */
 44   this._value = value['fraction'] ? value['integer'] + '.' + value['fraction'] : value['integer'];
 45   /**
 46    * @private
 47    * @type number
 48    */
 49   this._scale = value['fraction'].length;
 50 
 51   return this;
 52 };

 54  * @param {number} [scale]
 55  * @returns {number}
 56  * @private

 58 fNumber._fixScale = function (scale) {
 59   if (scale !== undefined) {
 60     scale = +scale;
 61     if (isNaN(scale) || scale < 0) {
 62       scale = 0;
 63     }
 64   }
 65   else {
 66     scale = 0;
 67   }
 68   return scale;
 69 };
 70 /**
 71  * @param {string} number
 72  * @returns {string}
 73  */
 74 fNumber._fixSign = function (number) {
 75   number = number.toString();
 76   if (number.charAt(0).match(/[0-9]/)) {
 77     number = '+' + number;
 78   }
 79   return number;
 80 };
 81 /**
 82  * Strips the leading zeroes off a number.
 83  * @private
 84  * @param {number|string} number The number to strip.
 85  * @returns {string} The number with leading zeroes stripped.
 86  */
 87 fNumber.stripLeadingZeroes = function (number) {
 88   number = String(number);
 89   var firstChar = number.charAt(0);
 90   var sign = '';
 91 
 92   if (!(parseInt(firstChar, 10) || firstChar === '.')) {
 93     sign = firstChar;
 94     number = number.substr(1);
 95   }
 96 
 97   number = fUTF8.ltrim(number);
 98   if (number.length === 0 || number.charAt(0) === '.') {
 99     number = '0' + number;
100   }
101 
102   return sign + number;
103 };
104 /**
105  * Checks to see if a number is equal to zero.
106  * @param {string} number The number to check, first character should be the sign.
107  * @returns {boolean} If the number is equal to zero.
108  * @private
109  */
110 fNumber.isZero = function (number) {
111   number = number.replace(/\./g, '').substr(1);
112   return fUTF8.trim(number, '0') === '';

114 /**
115  * Parses a number to ensure it is valid. Sets to 0 if invalid.
116  * @private
117  * @param {Object|string} number The number to parse.
118  * @param {string} element The element to return. One of: 'number', 'integer',
119  *   'fraction', 'array'. 'array' technically returns a JavaScript object.
120  * @returns {string|Object} The requested parsed element.
121  */
122 fNumber.parse = function (number, element) {
123   if (typeof number === 'object') {
124     number = number.toString();
125   }
126   else {
127     number = String(number);
128   }
129   number = fUTF8.trim(number);
130 
131   // has a D modifier in the original, but that should not be needed here
132   var matches = number.match(/^([+\-]?)((?:\d*\.)?\d+)(?:e([+\-]?)(\d+))?$/i);
133   if (!matches) {
134     // NOTE Normally throws an fValidationException here
135     number = '0';
136     matches = [number, '+'];
137   }
138 
139   var sign = (matches[1] === '-') ? '-' : '+';
140   number = fNumber.stripLeadingZeroes(matches[2]);
141 
142   var exponent = matches[4] ? matches[4] : null;
143   var exponentSign = matches[3] ? matches[3] : null;
144   var fraction, integer, decimalPosition, parts, output = {};
145 
146   if (exponent) {
147     if (exponentSign === '-') {
148       decimalPosition = fUTF8.pos(number, '.');
149 
150       if (decimalPosition === false) {
151         fraction = '';
152         integer = number;
153       }
154       else {
155         var fracSub = fUTF8.pos(number, '.');
156         var integerSub = parseInt(fUTF8.pos(number, '.'), 10);
157         if (!fracSub) {
158           fracSub = 0;
159         }
160         if (isNaN(integerSub)) {
161           integer = 0;
162         }
163 
164         fraction = number.substr(fracSub + 1);
165         integer = number.substr(0, integerSub);
166       }
167     }
168     else {
169       number += fUTF8.pad('', exponent, '0', 'right');
170     }
171   }
172 
173   number = fNumber.stripLeadingZeroes(number);
174 
175   if (fNumber.isZero(sign + number)) {
176     sign = '+';
177   }
178 
179   parts = (sign + number).split('.');
180 
181   output['integer'] = parts[0];
182   output['fraction'] = parts[1] ? parts[1] : '';
183 
184   if (element === 'integer' || element === 'number') {
185     return output['integer'];
186   }
187   else if (element === 'fraction') {
188     return output['fraction'];
189   }
190   else if (element === 'array') { // yes technically it's an Object
191     return output;
192   }
193 
194   // Falls through for element 'number'
195   return parts.join('.');
196 };

354     var joinObject = function (obj, sep) {
355       var ret = '';
356       var count = (function () {
357         var count = 0;
358         for (var key in obj) {
359           if (obj.hasOwnProperty(key)) {
360             count++;
361           }
362         }
363         return count;
364       })();
365 
366       var i = 0;
367       for (var key in obj) {
368         if (obj.hasOwnProperty(key) && i !== (count-1)) {
369           ret += obj[key] + sep;
370           i++;
371         }
372       }
373       return ret;
374     };

376     // This is to get rid of this warning: actual parameter 1 of joinObject does not match formal parameter
377     var populate = function (obj) {
378       var ret = {};
379       for (var key in obj) {
380         if (obj.hasOwnProperty(key)) {
381           ret[key] = obj[key];
382         }
383       }
384       return key;
385     };
386     var obj1 = populate(objNumber1), obj2 = populate(objNumber2);
387     number1 = joinObject(obj1, '.');
388     number2 = joinObject(obj2, '.');

391     number1 = objNumber1['integer'];

398   number2 = number2.charAt(0) + fUTF8.pad(number2.substr(1), len, '0', 'left');

429 /**
430  * Compares two numbers.
431  * @param {string} number1
432  * @param {string} number2
433  * @returns {number} Less than 0 if <code>number1</code> is less than
434  *   <code>number2</code>, 0 if equal, greater than 0 if <code>number1</code>
435  *   is greater than <code>number2</code>
436  * @private
437  */
438 fNumber._cmp = function (number1, number2) {
439   number1 = fNumber._fixSign(number1);
440   number2 = fNumber._fixSign(number2);
441 
442   if (number1.charAt(0) !== number2.charAt(0)) {
443     return number1.charAt(0) === '+' ? 1 : -1;
444   }
445 
446   if (number1.charAt(0) === '-') {
447     return fUTF8.natcmp(number2.substr(1), number1.substr(1));
448   }
449 
450   return fUTF8.natcmp(number1.substr(1), number2.substr(1));
451 };
452 /**
453  * Subtracts numbers.
454  * @param {string} minuend The number to subract from.
455  * @param {string} subtrahend The number to subtract.
456  * @param {number} [scale] The number of digits after the decimal.
457  * @returns {string} The difference.
458  * @private
459  */
460 fNumber._performSub = function (minuend, subtrahend, scale) {
461   var array = fNumber._normalize(minuend, subtrahend), diff;
462   minuend = array[0];
463   subtrahend = array[1];
464   scale = fNumber._fixScale(scale);
465 
466   if (subtrahend.charAt(0) === '-') {
467     if (minuend.charAt(0) === '-') {
468       return fNumber._performSub('+' + subtrahend.substr(1), '+' + minuend.substr(1), scale);
469     }
470 
471     return fNumber._performAdd('+' + subtrahend.substr(1), '+' + minuend.substr(1), scale);
472   }
473 
474   if (minuend.charAt(0) === '-') {
475     var sum = fNumber._performAdd('+' + minuend.substr(1), subtrahend, scale);
476     return '-' + sum.substr(1);
477   }
478 
479   if (fNumber._cmp(minuend, subtrahend) < 0) {
480     diff = fNumber._performSub('+' + subtrahend.substr(1), '+' + minuend.substr(1), scale);
481     diff = diff.substr(1);
482     return minuend.charAt(0) === '+' ? '-' + diff : '+' + diff;
483   }
484 
485   var borrow = 0;
486   var output = '';
487 
488   for (var i = minuend.length - 1; i > 0; i--) {
489     if (minuend.charAt(i) === '.') {
490       output = '.' + output;
491       continue;
492     }
493 
494     diff = (+minuend.charAt(i)) - (+subtrahend.charAt(i)) - borrow;
495     borrow = 0;
496     while (diff < 0) {
497       borrow++;
498       diff += 10;
499     }
500     output = diff + output;
501   }
502 
503   output = fNumber._setScale(minuend.charAt(0) + output, scale);
504 
505   return fNumber.stripLeadingZeroes(output);
506 };
507 /**
508  * Adds two numbers together.
509  * @param {string} number1 The first addend.
510  * @param {string} number2 The second addend.
511  * @param {number} [scale] The number of digits after the decimal.
512  * @returns {string} The sum of the two numbers.
513  * @private
514  */
515 fNumber._performAdd = function (number1, number2, scale) {
516   if (scale === undefined) {
517     scale = 0;
518   }
519 
520   var array = fNumber._normalize(number1, number2);
521   number1 = array[0];
522   number2 = array[1];
523 
524   if (fNumber.isZero(number1)) {
525     return fNumber._setScale(number2, scale);
526   }
527   else if (fNumber.isZero(number2)) {
528     return fNumber._setScale(number1, scale);
529   }
530 
531   if (number1.charAt(0) !== number2.charAt(0)) {
532     if (number1.charAt(0) === '-') {
533       return fNumber._performSub(number2, '+' + number1.substr(1));
534     }
535     return fNumber._performSub(number2, '+' + number2.substr(1));
536   }
537 
538   var carry = 0;
539   var output = '';
540   var sum;
541 
542   for (var i = number1.length - 1; i > 0; i--) {
543     if (number1.charAt(i) === '.') {
544       output = '.' + output;
545       continue;
546     }
547 
548     sum = String((+number1.charAt(i)) + (+number2.charAt(i)) + carry);
549     carry = sum.length > 1 ? sum.substr(0, -1) : 0;
550     output = sum.substr(-1) + output;
551   }
552 
553   if (carry) {
554     output = carry + output;
555   }
556 
557   output = fNumber._setScale(number1.charAt(0) + output, scale);
558 
559   return fNumber.stripLeadingZeroes(output);
560 };
561 /**
562  * Adds two numbers together.
563  * @param {fNumber|number|string} addend The addend
564  * @param {number} [scale] The number of places after the decimal.
565  * @returns {fNumber} The sum.
566  */
567 fNumber.prototype.add = function (addend, scale) {
568   scale = fNumber._fixScale(scale);
569   addend = fNumber.parse(addend.toString(), 'number').toString();
570 
571   // Not using bcadd
572   var sum = fNumber._performAdd(this._value, addend, scale);
573 
574   return new fNumber(sum, scale);
575 };

1015

1034 /**
1035  * Rounds the number.
1036  * @param {number} scale The number of places after (or before if negative) the
1037  *   decimal to round to.
1038  * @returns {fNumber} The rounded result.
1039  */
1040 fNumber.prototype.round = function (scale) {
1041   scale = fNumber._fixScale(scale);
1042 
1043   var number = fNumber._setScale(this._value, (scale < 0) ? 1 : scale + 1);
1044   var length = number.length;
1045   var add = false;
1046 
1047   if (scale === 0) {
1048     if (number.charAt(length - 1) >= 5) {
1049       add = '1';
1050     }
1051     number = number.substr(0, -2);
1052   }
1053   else if (scale > 0) {
1054     if (number.charAt(length - 1) >= 5) {
1055       add = '0.' + fUTF8.pad('', scale-1, '0') + '1';
1056     }
1057     number = number.substr(0, -1);
1058   }
1059   else {
1060     number = number.substr(0, number.indexOf('.'));
1061 
1062     if (Math.abs(scale) >= number.length) {
1063       number = '0';
1064     }
1065     else {
1066       if (number.charAt(number.length - Math.abs(scale)) >= 5) {
1067         add = '1' + fUTF8.pad('', Math.abs(scale), '0');
1068       }
1069       number = number.substr(0, scale);
1070       number += fUTF8.pad('', Math.abs(scale), '0');
1071     }
1072   }
1073 
1074   if (add) {
1075     number = fNumber._performAdd(number, add.toString(), scale);
1076   }
1077 
1078   return new fNumber(number);
1079 };
