  6 /**
  7  * This function returns a string with whitespace stripped from the beginning
  8  *   and end of str. Without the second parameter, trim() will strip these
  9  *   characters:
 10  * <ul>
 11  * <li>" " (ASCII 32 (0x20)), an ordinary space.</li>
 12  * <li>"\t" (ASCII 9 (0x09)), a tab.</li>
 13  * <li>"\n" (ASCII 10 (0x0A)), a new line (line feed).</li>
 14  * <li>"\r" (ASCII 13 (0x0D)), a carriage return.</li>
 15  * <li>"\0" (ASCII 0 (0x00)), the NUL-byte.</li>
 16  * <li>"\x0B" (ASCII 11 (0x0B)), a vertical tab.</li>
 17  * </ul>
 18  * @param {string} str The string that will be trimmed.
 19  * @param {string} [charlist] Optionally, the stripped characters can also be
 20  *   specified using the charlist parameter. Simply list all characters that
 21  *   you want to be stripped in a string.
 22  * @returns {string} The trimmed string.
 23  */
 24 fUTF8.trim = function (str, charlist) {
 25     if (str.trim && !charlist) {
 26       return str.trim();
 27     }
 28 
 29     // http://kevin.vanzonneveld.net
 30     // +   original by: Kevin van Zonneveld (http://kevin.vanzonneveld.net)
 31     // +   improved by: mdsjack (http://www.mdsjack.bo.it)
 32     // +   improved by: Alexander Ermolaev (http://snippets.dzone.com/user/AlexanderErmolaev)
 33     // +      input by: Erkekjetter
 34     // +   improved by: Kevin van Zonneveld (http://kevin.vanzonneveld.net)
 35     // +      input by: DxGx
 36     // +   improved by: Steven Levithan (http://blog.stevenlevithan.com)
 37     // +    tweaked by: Jack
 38     // +   bugfixed by: Onno Marsman
 39     // *     example 1: trim('    Kevin van Zonneveld    ');
 40     // *     returns 1: 'Kevin van Zonneveld'
 41     // *     example 2: trim('Hello World', 'Hdle');
 42     // *     returns 2: 'o Wor'
 43     // *     example 3: trim(16, 1);
 44     // *     returns 3: 6
 45     var whitespace, l = 0,
 46         i = 0;
 47 
 48     if (!charlist) {
 49         // default list
 50         whitespace = " \n\r\t\f\x0b\xa0\u2000\u2001\u2002\u2003\u2004\u2005\u2006\u2007\u2008\u2009\u200a\u200b\u2028\u2029\u3000";
 51     } else {
 52         // preg_quote custom list
 53         charlist += '';
 54         whitespace = charlist.replace(/([\[\]\(\)\.\?\/\*\{\}\+\$\^\:])/g, '$1');
 55     }
 56 
 57     l = str.length;
 58     for (i = 0; i < l; i++) {
 59         if (whitespace.indexOf(str.charAt(i)) === -1) {
 60             str = str.substring(i);
 61             break;
 62         }
 63     }
 64 
 65     l = str.length;
 66     for (i = l - 1; i >= 0; i--) {
 67         if (whitespace.indexOf(str.charAt(i)) === -1) {
 68             str = str.substring(0, i + 1);
 69             break;
 70         }
 71     }
 72 
 73     return whitespace.indexOf(str.charAt(0)) === -1 ? str : '';
 74 };
 75 /**

 77  * Without the second parameter, ltrim() will strip these characters:
 78  * <ul>

 93   if (str.trimLeft && !charlist) {

109 /**
110  * Pad a string to a certain length with another string.
111  * @param {string} input The input string.
112  * @param {number} pad_length The character length to pad the string to.
113  * @param {string} [pad_string= ] The string to pad the source string with.
114  * @param {string} [pad_type='right'] One of: 'left', 'right', 'both'.
115  * @returns {string} Returns the padded string.
116  */
117 fUTF8.pad = function (input, pad_length, pad_string, pad_type) {
118     // http://kevin.vanzonneveld.net
119     // +   original by: Kevin van Zonneveld (http://kevin.vanzonneveld.net)
120     // + namespaced by: Michael White (http://getsprink.com)
121     // +      input by: Marco van Oort
122     // +   bugfixed by: Brett Zamir (http://brett-zamir.me)
123     // *     example 1: str_pad('Kevin van Zonneveld', 30, '-=', 'STR_PAD_LEFT');
124     // *     returns 1: '-=-=-=-=-=-Kevin van Zonneveld'
125     // *     example 2: str_pad('Kevin van Zonneveld', 30, '-', 'STR_PAD_BOTH');
126     // *     returns 2: '------Kevin van Zonneveld-----'
127     var half = '',
128         pad_to_go;
129 
130     /**
131      * @private
132      * @param {string} s
133      * @param {number} len
134      * @returns {string}
135      */
136     var str_pad_repeater = function (s, len) {
137         var collect = '',
138             i;
139 
140         while (collect.length < len) {
141             collect += s;
142         }
143         collect = collect.substr(0, len);
144 
145         return collect;
146     };
147 
148     input += '';
149     pad_string = pad_string !== undefined ? pad_string : ' ';
150 
151     if (pad_type != 'left' && pad_type != 'right' && pad_type != 'both') {
152         pad_type = 'right';
153     }
154     if ((pad_to_go = pad_length - input.length) > 0) {
155         if (pad_type == 'left') {
156             input = str_pad_repeater(pad_string, pad_to_go) + input;
157         } else if (pad_type == 'right') {
158             input = input + str_pad_repeater(pad_string, pad_to_go);
159         } else if (pad_type == 'both') {
160             half = str_pad_repeater(pad_string, Math.ceil(pad_to_go / 2));
161             input = half + input + half;
162             input = input.substr(0, pad_length);
163         }
164     }
165 
166     return input;
167 };

228 /**
229  * Compares strings using a natural order algorithm.<br>
230  * License: LGPL
231  * @param {string} a The first string.
232  * @param {string} b The second string.
233  * @returns {number} Less than 0 if f_string1 is less than f_string2, greater
234  *   than 0 if f_string1 is f_string2, and 0 if they are equal.
235  * @see http://www.davekoelle.com/files/alphanum.js
236  */
237 fUTF8.natcmp = function (a, b) {
238   function chunkify(t) {
239     var tz = [], x = 0, y = -1, n = 0, i, j;
240 
241     while ((i = (j = t.charAt(x++)).charCodeAt(0))) {
242       var m = (i == 46 || (i >=48 && i <= 57));
243       if (m !== n) {
244         tz[++y] = "";
245         n = m;
246       }
247       tz[y] += j;
248     }
249     return tz;
250   }
251 
252   var aa = chunkify(a);
253   var bb = chunkify(b);
254 
255   for (var x = 0; aa[x] && bb[x]; x++) {
256     if (aa[x] !== bb[x]) {
257       var c = Number(aa[x]), d = Number(bb[x]);
258       if (c == aa[x] && d == bb[x]) {
259         return c - d;
260       } else return (aa[x] > bb[x]) ? 1 : -1;
261     }
262   }
263 
264   return aa.length - bb.length;
265 };

395 /**
396  * Finds the last position (in characters) of the search value in the string.
397  * @param {string} haystack String to search in.
398  * @param {string} needle String to search for.
399  * @param {number} [offset=0] The character position to start searching from.
400  * @returns {number|boolean} The last position of the character, or boolean
401  *   <code>false</code> if it is not found.
402  */
403 fUTF8.rpos = function (haystack, needle, offset) {
404   // If all three are false-like values, return false
405   if (!haystack && !needle && !offset) {
406     return false;
407   }
408 
409   // If the needle is a false-like value, return false
410   if (!needle) {
411     return false;
412   }
413 
414   var i = -1;
415 
416   if (offset !== undefined && offset > 0) {
417     i = haystack.slice(offset).lastIndexOf(needle);
418     if (i !== -1) {
419       i += offset;
420     }
421   }
422   else if (offset < 0) {
423     i = haystack.slice(Math.abs(offset)).lastIndexOf(needle) + 1;
424   }
425   else {
426     i = haystack.lastIndexOf(needle);
427   }
428 
429   return i >= 0 ? i : false;
430 };

474 /**
475  * Converts a UTF-8 character to a unicode code point. Based on phpjs version.
476  * @param {string} string The character to decode.
477  * @returns {string} The U+hex unicode code point for the character.
478  * @see http://phpjs.org/functions/ord:483
479  */
480 fUTF8.ord = function (string) {
481     // http://kevin.vanzonneveld.net
482     // +   original by: Kevin van Zonneveld (http://kevin.vanzonneveld.net)
483     // +   bugfixed by: Onno Marsman
484     // +   improved by: Brett Zamir (http://brett-zamir.me)
485     // +   input by: incidence
486     // *     example 1: ord('K');
487     // *     returns 1: 75
488     // *     example 2: ord('\uD800\uDC00'); // surrogate pair to create a single Unicode character
489     // *     returns 2: 65536
490     var str = string,
491         code = str.charCodeAt(0);
492     if (0xD800 <= code && code <= 0xDBFF) { // High surrogate (could change last hex to 0xDB7F to treat high private surrogates as single characters)
493         var hi = code;
494         if (str.length === 1) {
495             return 'U+' + fUTF8.pad(code.toString(), 4, '0', 'left'); // This is just a high surrogate with no following low surrogate, so we return its value;
496             // we could also throw an error as it is not a complete character, but someone may want to know
497         }
498         var low = str.charCodeAt(1);
499         return 'U+' + fUTF8.pad((((hi - 0xD800) * 0x400) + (low - 0xDC00) + 0x10000).toString(), 4, '0', 'left');
500     }
501     if (0xDC00 <= code && code <= 0xDFFF) { // Low surrogate
502         return 'U+' + fUTF8.pad(code.toString(), 4, '0', 'left'); // This is just a low surrogate with no preceding high surrogate, so we return its value;
503         // we could also throw an error as it is not a complete character, but someone may want to know
504     }
505     return 'U+' + fUTF8.pad(code.toString(), 4, '0', 'left');
506 };

582

585   }

597 /**
598  * Converts the first character of every word to upper-case. Currently, this
599  *   only works for ASCII strings.<br>
600  * Thanks to mauke in ##javascript for the regular expression.
601  * @param {string} str The string to process.
602  * @returns {string} The processed string.
603  */
604 fUTF8.ucwords = function (str) {
605   // JavaScript seems to have less problems with this compared to PHP, so this should work most of the time
606   // 'ⓔ ἡ ᾥ ａ ჴ ց ր' -> Ⓔ Ἡ ᾭ Ａ ჴ Ց Ր
607   // Can only be the first character
608   var special = {
609     '(': '(',
610     '"': '"',
611     ': "'",
612     '‘': '‘',
613     '“': '“'
614   };
615   var regex = /\w[’']\w|\b\w/g;
616 
617   return str.replace(regex, function (m0) {
618     if (m0.length > 1) {
619       return m0;
620     }
621     return m0.toUpperCase();
622   });
623 };
624 /**

635 fUTF8.wordwrap = function (str, width, breakChar, cut) {
636     // http://kevin.vanzonneveld.net

646     // *     returns 2: 'The quick brown fox <br />\njumped over the lazy<br />\n dog.'
647     // *     example 3: wordwrap('Lorem ipsum dolor sit amet, consectetur adipisicing elit, sed do eiusmod tempor incididunt ut labore et dolore magna aliqua. Ut enim ad minim veniam, quis nostrud exercitation ullamco laboris nisi ut aliquip ex ea commodo consequat.');

649     // PHP Defaults
650     var m = ((arguments.length >= 2) ? arguments[1] : 75);
651     var b = ((arguments.length >= 3) ? arguments[2] : "\n");
652     var c = ((arguments.length >= 4) ? arguments[3] : false);

665         }
666     }
667
