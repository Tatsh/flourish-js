1 /**
  2  * Provides notation conversion support.
  3  * @constructor
  4  */
  5 var fGrammar = function () {};
  6 /**
  7  * Cache of strings that have been humanized.
  8  * @type Object
  9  * @private
 10  */
 11 fGrammar._humanizeCache = {};
 12 /**
 13  * Cache of strings that have been underscorized.
 14  * @type Object
 15  * @private
 16  */
 17 fGrammar._underscorizeCache = {};
 18 /**
 19  * Cache of strings that have been converted to camel case.
 20  * @type Object
 21  * @private
 22  */
 23 fGrammar._camelizeCache = {
 24   upper: {},
 25   lower: {}
 26 };
 27 /**
 28  * This method is marked private, but it is intended to be used by extensions
 29  *   such as sGrammar.
 30  * @private
 31  * @param {string} str
 32  * @param {string} delimiter
 33  * @param {Object} cacheToCheck
 34  * @returns {string}
 35  */
 36 fGrammar._commonize = function (str, delimiter, cacheToCheck) {
 37   if (cacheToCheck[str]) {
 38     return cacheToCheck[str];
 39   }
 40 
 41   var original = str;
 42   str = str.charAt(0).toLowerCase() + str.substr(1);
 43 
 44   if (str.indexOf('_') !== -1 && str.toLowerCase() === str) {
 45   }
 46   else if (str.indexOf(' ') !== -1) {
 47     str = str.replace(/\s+/g, '_').toLowerCase();
 48   }
 49   else {
 50     var old;
 51     do {
 52       old = str;
 53       str = str.replace(/([a-zA-Z])([0-9])/, '$1' + delimiter + '$2');
 54       str = str.replace(/([a-z0-9A-Z])([A-Z])/, '$1' + delimiter + '$2');
 55     } while (old !== str);
 56 
 57     str = str.toLowerCase();
 58   }
 59 
 60   cacheToCheck[original] = str;
 61 
 62   return str;
 63 };
 64 /**
 65  * Converts a <code>camelCase</code>, human-friendly or
 66  *   <code>underscore_notation</code> string to
 67  *   <code>underscore_notation</code>.
 68  * @param {string} str String to underscorise.
 69  * @returns {string} String, underscorised.
 70  */
 71 fGrammar.underscorize = function (str) {
 72   return fGrammar._commonize(str, '_', fGrammar._underscorizeCache);
 73 };
 74 /**
 75  * Makes an <code>underscore_notation</code>, <code>camelCase</code>, or
 76  *   human-friendly string into a human-friendly string.
 77  * @param {string} str String to 'humanise'.
 78  * @returns {string} String, 'humanised.'
 79  */
 80 fGrammar.humanize = function (str) {
 81   if (fGrammar._humanizeCache[str]) {
 82     return fGrammar._humanizeCache[str];
 83   }
 84 
 85   var original = str;
 86 
 87   if (str.indexOf(' ') === -1 && str.indexOf('_') !== -1) {
 88     str = fGrammar.underscorize(str);
 89   }
 90 
 91   //TODO Check that this part works; test!
 92   var regex = /(\b(api|css|gif|html|id|jpg|js|mp3|pdf|php|png|sql|swf|url|xhtml|xml)\b|\b\w)/;
 93   str = str.replace(/_/g, ' ').replace(regex, function (str, p1) {
 94     return p1.charAt(0).toUpperCase();
 95   });
 96 
 97   fGrammar._humanizeCache[original] = str;
 98 
 99   return str;
100 };
101 /**
102  * Converts an <code>underscore_notation</code>, human-friendly or
103  *   <code>camelCase</code> string to <code>camelCase</code>.
104  * @param {string} str String to convert.
105  * @param {boolean} [upper=false] If the camel case should be
106  *   <code>UpperCamelCase</code>.
107  * @param {string} [delimiter] Force a delimiter to be used.
108  * @returns {string} The converted string.
109  */
110 fGrammar.camelize = function (str, upper, delimiter) {
111   if (upper && fGrammar._camelizeCache.upper[str]) {
112     return fGrammar._camelizeCache.upper[str];
113   }
114   else if (!upper && fGrammar._camelizeCache.lower[str]) {
115     return fGrammar._camelizeCache.lower[str];
116   }
117 
118   if (upper === undefined) {
119     upper = false;
120   }
121   if (delimiter === undefined) {
122     delimiter = '_';
123   }
124 
125   var original = str;
126 
127   if (str.indexOf(' ') !== -1) {
128     str = str.replace(/\s+/g, delimiter).toLowerCase();
129   }
130 
131   if (str.indexOf(delimiter) === -1) {
132     if (upper) {
133       str = str.charAt(0).toUpperCase() + str.substr(1);
134     }
135   }
136   else {
137     var next = str.indexOf(delimiter);
138     var beginning;
139     var parts = [];
140     while (next !== -1) {
141       beginning = str.charAt(0).toUpperCase();
142       parts.push(beginning + str.substr(1, next - 1));
143       str = str.substr(next + 1);
144       next = str.indexOf(delimiter);
145     }
146 
147     // Last part of the string
148     parts.push(str.charAt(0).toUpperCase() + str.substr(1));
149 
150     str = parts.join('');
151 
152     if (!upper) {
153       str = str.charAt(0).toLowerCase() + str.substr(1);
154     }
155   }
156 
157   if (upper) {
158     fGrammar._camelizeCache.upper[original] = str;
159   }
160   else {
161     fGrammar._camelizeCache.lower[original] = str;
162   }
163 
164   return str;
165 };
166 /**
167  * Returns the singular or plural form of the word or based on the quantity
168  *   specified.
169  * @param {number} number The quantity.
170  * @param {string} singular The string to be returned for when <code>number =
171  *   1</code>.
172  * @param {string} plural The string to be returned for when
173  *   <code>number != 1</code>; use <code>%d</code> to place the quantity in the
174  *   string.
175  * @return {string} The singular or plural form of the word or based on the
176  *   quantity specified.
177  */
178 fGrammar.inflectOnQuantity = function (number, singular, plural) {
179   if (number === 0 || number > 1) {
180     return plural.replace('%d', String(number));
181   }
182   return singular;
183 };
