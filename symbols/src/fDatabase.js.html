  4 /**
  5  * Front end to IndexedDB. For all methods here, if the browser lacks support,
  6  *   they will fail silently.
  7  * @constructor
  8  * @param {string} name Database name to open.
  9  * @param {string} [version] Version to open. Recommended but not required.
 10  * @returns {fDatabase} The database object.
 11  */
 12 var fDatabase = function (name, version) {
 13   if (version === undefined) {
 14     version = '1.0';
 15   }
 16 
 17   /**
 18    * Database name.
 19    * @type string
 20    * @private
 21    */
 22   this._name = name;
 23   /**
 24    * Database version.
 25    * @type string
 26    * @private
 27    */
 28   this._version = version;
 29   /**
 30    * If connected.
 31    * @type boolean
 32    * @private
 33    */
 34   this._isConnected = false;
 35   /**
 36    * The indexed database.
 37    * @type (IDBDatabase|null)
 38    * @private
 39    */
 40   this._database = null;
 41   /**
 42    * The error handler.
 43    * @type function(Error)
 44    */
 45   this._errorHandler = fDatabase._defaultErrorHandler;
 46 
 47   return this;
 48 };
 49 /**
 50  * If the browser supports indexedDB.
 51  * @type boolean
 52  * @private
 53  */
 54 fDatabase._supportsIndexedDB = (function () {
 55   var possible = ['webkitIndexedDB', 'mozIndexedDB', 'indexedDB'];
 56   for (var i = 0; i < possible.length; i++) {
 57     if (possible[i] in window) {
 58       try {
 59         window.indexedDB = window[possible[i]];
 60       }
 61       catch (e) {}
 62       return true;
 63     }
 64   }
 65   return false;
 66 })();
 67 /**
 68  * Default failure handler.
 69  * @param {Error|Object} [e] The error.
 70  */
 71 fDatabase._defaultErrorHandler = function (e) {
 72   if (typeof e !== 'object') {
 73     e = {value: 'Unknown error'};
 74   }
 75   fCore.debug('%s', e.value);
 76 };
 77 /**
 78  * Set the database.
 79  * @param {IDBDatabase} database Database.
 80  * @returns {fDatabase} The object to allow method chaining.
 81  */
 82 fDatabase.prototype.setDatabase = function (database) {
 83   if (!fDatabase._supportsIndexedDB) {
 84     return this;
 85   }
 86 
 87   this._database = database;
 88   return this;
 89 };
 90 /**
 91  * Alias to setDatabase() for compliance.
 92  * @param {IDBDatabase} database Database.
 93  * @returns {fDatabase} The object to allow method chaining.
 94  */
 95 fDatabase.prototype.setConnection = function (database) {
 96   return this.setDatabase(database);
 97 };

133 /**
134  * Connect to the database. All methods that need to connect will call this,
135  *   so calling this before doing anything is not required.
136  * @param {function(fDatabase)} cb Callback for when the database is opened.
137  * @returns {fDatabase} The object to allow method chaining.
138  */
139 fDatabase.prototype.connect = function (cb) {
140   if (!fDatabase._supportsIndexedDB) {
141     return this;
142   }
143 
144   if (!this._isConnected) {
145     var request = window.indexedDB.open(this._name);
146     var instance = this;
147 
148     request.onsuccess = function (event) {
149       instance.setDatabase(event.target.result);
150       instance._isConnected = true;
151 
152       var db = instance.getConnection();
153       var version = instance.getVersion();
154 
155       if (version !== db.version) {
156         var versionRequest = db.setVersion(version);
157 
158         versionRequest.onfailure = instance.getErrorHandler();
159         versionRequest.onsuccess = function () {
160           cb(instance);
161         };
162       }
163       else {
164         cb(instance);
165       }
166     };
167     request.onfailure = this._errorHandler;
168   }
169   else {
170     cb(this);
171   }
172 
173   return this;
174 };
