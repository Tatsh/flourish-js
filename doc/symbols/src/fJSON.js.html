174 /**
175  * @private
176  * @param {number|null} last
177  * @param {string} element
178  * @returns {number}
179  */
180 fJSON.getElementType = function (last, element) {
181   if (element === '[') {
182     return fJSON.J_ARRAY_OPEN;
183   }
184   if (element === ']') {
185     return fJSON.J_ARRAY_CLOSE;
186   }
187   if (element === '{') {
188     return fJSON.J_OBJ_OPEN;
189   }
190   if (element === '}') {
191     return fJSON.J_OBJ_CLOSE;
192   }
193   if (!isNaN(parseInt(element, 10))) {
194     return fJSON.J_INTEGER;
195   }
196   if (!isNaN(parseFloat(element))) {
197     return fJSON.J_FLOAT;
198   }
199   if (element === 'true') {
200     return fJSON.J_TRUE;
201   }
202   if (element === 'false') {
203     return fJSON.J_FALSE;
204   }
205   if (element === 'null') {
206     return fJSON.J_NULL;
207   }
208 
209   var lastStack = fJSON._stack[fJSON._stack.length - 1];
210   if (element === ',' && lastStack[0] === fJSON.J_ARRAY_OPEN) {
211     return fJSON.J_ARRAY_COMMA;
212   }
213 
214   if (element === ',') {
215     return fJSON.J_OBJ_COMMA;
216   }
217 
218   if (element === ':') {
219     return fJSON.J_COLON;
220   }
221 
222   if (last === fJSON.J_OBJ_OPEN || last === fJSON.J_OBJ_COMMA) {
223     return fJSON.J_KEY;
224   }
225 
226   return fJSON.J_STRING;
227 };
228 /**
229  * @param {number} type
230  * @param {string} element
231  * @private
232  * @returns {string|number|null|boolean}
233  */
234 fJSON.scalarize = function (type, element) {
235   var ret = null;
236 
237   if (type === fJSON.J_INTEGER) {
238     ret = parseInt(element, 10);
239     if (isNaN(ret)) {
240       ret = 0;
241     }
242   }
243   if (type === fJSON.J_FLOAT) {
244     ret = parseFloat(ret);
245     if (isNaN(ret)) {
246       ret = 0;
247     }
248   }
249   if (type === fJSON.J_FALSE) {
250     ret = false;
251   }
252   if (type === fJSON.J_TRUE) {
253     ret = true;
254   }
255   if (type === fJSON.J_STRING || type === fJSON.J_KEY) {
256     ret = element.substr(1, element.length - 2);
257 
258     for (var key in fJSON.controlCharacterMap) {
259       ret = ret.replace(new RegExp(fJSON.controlCharacterMap[key], 'g'), key);
260     }
261 
262     ret = ret.replace(/\\\\u([0-9a-fA-F]{4})/, function (m0, m1) {
263       return fUTF8.chr(m1);
264     });
265   }
266 
267   return ret;
268 };
269 /**
270  * Decode a JSON string.
271  * @param {string} json JSON string to decode.
272  * @param {boolean} [useNative=true] Pass <code>false</code> to force not using
273  *   <code>window.JSON</code> (if it is available).
274  * @returns {*} An object, array, string, number, or null. Can also return
275  *   undefined. However, if the non-native decoder is used, only the mentioned
276  *   types can be returned.
277  */
278 fJSON.decode = function (json, useNative) {
279   if (json === undefined) {
280     return null;
281   }
282   if (useNative === undefined) {
283     useNative = true;
284   }
285 
286   var type;
287 
288   if (window.JSON && window.JSON.parse && useNative) {
289     try {
290       return window.JSON.parse(json);
291     }
292     catch (e) {
293       return null;
294     }
295   }
296 
297   fJSON._stack = [];
298 
299   var intNumber = parseInt(json, 10);
300   var floatNumber = parseFloat(json);
301   if (intNumber.toString() === json) {
302     return intNumber;
303   }
304   else if (floatNumber.toString() === json) {
305     return floatNumber;
306   }
307 
308   json = fUTF8.trim(json);
309 
310   if (json === '') {
311     return null;
312   }
313 
314   var matchedLength = 0, matches;
315   var last = null;
316   var lastKey = null;
317   var output = null;
318   var container = null;
319   var refMatch;
320   var newContainer;
321   var stackEnd;
322 
323 /*
324     preg_match_all('~\[|                   # Array begin
325              \]|                   # Array end
326              {|                    # Object begin
327              }|                    # Object end
328              -?(?:0|[1-9]\d*)                        # Float
329                (?:\.\d*(?:[eE][+\-]?\d+)?|
330                (?:[eE][+\-]?\d+))|
331              -?(?:0|[1-9]\d*)|             # Integer
332              true|                   # True
333              false|                    # False
334              null|                   # Null
335              ,|                    # Member separator for arrays and objects
336              :|                    # Value separator for objects
337              "(?:(?:(?!\\\\u)[^\\\\"\n\b\f\r\t]+)|   # String
338                \\\\\\\\|
339                \\\\/|
340                \\\\"|
341                \\\\b|
342                \\\\f|
343                \\\\n|
344                \\\\r|
345                \\\\t|
346                \\\\u[0-9a-fA-F]{4})*"|
347              \s+                   # Whitespace
348              ~x', $json, $matches);
349 */
350 
351   // IE 7 has trouble with escaped '/' which is \/, so replace all real \ with nothing
352   if (json.indexOf('\/')) {
353     json = json.replace(/\\/g, '');
354   }
355 
356   // Ugly, just combined version of the above
357   matches = json.match(/\[|\]|\{|\}|-?(?:0|[1-9]\d*)(?:\.\d*(?:[eE][+\-]?\d+)?|(?:[eE][+\-]?\d+))|-?(?:0|[1-9]\d*)|true|false|null|,|\:|"(?:(?:(?!\\\\u)[^\\\\"\n\b\f\r\t]+)|\\\\\\\\|\\\\\/|\\\\"|\\\\b|\\\\f|\\\\n|\\\\r|\\\\t|\\\\u[0-9a-fA-F]{4})*"|\s+/g);
358 
359   // JSON is just a simple string
360   if (matches === null) {
361     return json;
362   }
363 
364   // IE7 creates blanks for some reason
365   var replace = [];
366   for (var i = 0; i < matches.length; i++) {
367     if (matches[i] !== '') {
368       replace.push(matches[i]);
369     }
370   }
371   matches = replace;
372 
373   if (matches[0] !== '[' && matches[0] !== '{') {
374     return null; // invalid JSON
375   }
376 
377   var match;
378   for (i = 0; i < matches.length; i++) {
379     match = matches[i];
380 
381     if (matchedLength === 0) {
382       if (match === '[') {
383         output = [];
384         last = fJSON.J_ARRAY_OPEN;
385       }
386       else {
387         output = {};
388         last = fJSON.J_OBJ_OPEN;
389       }
390       fJSON._stack.push([last, output]);
391       container = output;
392       matchedLength = 1;
393       continue;
394     }
395 
396     matchedLength += match.length;
397 
398     // Skip over white space
399     if (fUTF8.trim(match) === '') {
400       continue;
401     }
402 
403     type = fJSON.getElementType(last, match);
404 
405     // Invalid JSON
406     if (fJSON.nextValues[last][type] === undefined) {
407       break;
408     }
409 
410     // Decode the data values
411     match = fJSON.scalarize(type, match);
412 
413     if (type === fJSON.J_COLON ||
414         type === fJSON.J_OBJ_COMMA ||
415         type === fJSON.J_ARRAY_COMMA ||
416         type === fJSON.J_KEY) {
417       last = type;
418       if (type === fJSON.J_KEY) {
419         lastKey = match;
420       }
421       continue;
422     }
423 
424     // Flag used to indicate if an or object is being added
425     refMatch = false;
426 
427     // Close an object or array
428     if (type === fJSON.J_OBJ_CLOSE || type === fJSON.J_ARRAY_CLOSE) {
429       fJSON._stack.pop();
430       if (fJSON._stack.length === 0) {
431         break;
432       }
433 
434       newContainer = fJSON._stack[fJSON._stack.length - 1];
435       container = newContainer[1];
436       last = type;
437       continue;
438     }
439 
440     // Opening a new object
441     if (type === fJSON.J_OBJ_OPEN) {
442       match = {};
443       refMatch = true;
444     }
445     if (type === fJSON.J_ARRAY_OPEN) {
446       match = [];
447       refMatch = true;
448     }
449 
450     if (refMatch) {
451       fJSON._stack.push([type, match]);
452       stackEnd = fJSON._stack[fJSON._stack.length - 1];
453     }
454 
455     if (last === fJSON.J_COLON) {
456       if (refMatch) {
457         container[lastKey] = stackEnd[1];
458         container = stackEnd[1];
459       }
460       else {
461         container[lastKey] = match;
462       }
463     }
464     else {
465       if (refMatch) {
466         container.push(stackEnd[1]);
467         container = stackEnd[1];
468       }
469       else {
470         container.push(match);
471       }
472     }
473 
474     if (last === fJSON.J_COLON) {
475       lastKey = null;
476     }
477     last = type;
478   }
479 
480   if (matchedLength !== json.length || fJSON._stack.length > 0) {
481     return null;
482   }
483 
484   return output;
485 };
486 /**
487  * Encode a JavaScript object to a JSON string.
488  * @param {Array|Object|number|string} mixed_val Value to encode.
489  * @returns {string|null} JSON string or <code>null</code>.
490  */
491 fJSON.encode = function (mixed_val) {
492     // http://kevin.vanzonneveld.net
493     // +      original by: Public Domain (http://www.json.org/json2.js)
494     // + reimplemented by: Kevin van Zonneveld (http://kevin.vanzonneveld.net)
495     // +      improved by: Michael White
496     // +      input by: felix
497     // +      bugfixed by: Brett Zamir (http://brett-zamir.me)
498     // *        example 1: json_encode(['e', {pluribus: 'unum'}]);
499     // *        returns 1: '[\n    "e",\n    {\n    "pluribus": "unum"\n}\n]'
500 /*
501         http://www.JSON.org/json2.js
502         2008-11-19
503         Public Domain.
504         NO WARRANTY EXPRESSED OR IMPLIED. USE AT YOUR OWN RISK.
505         See http://www.JSON.org/js.html
506     */
507     var retVal, json = window.JSON;
508     try {
509         if (typeof json === 'object' && typeof json.stringify === 'function') {
510             retVal = json.stringify(mixed_val); // Errors will not be caught here if our own equivalent to resource
511             if (retVal === undefined) {
512                 //throw new SyntaxError('json_encode');
513                 fCore.debug('Native JSON encoder returned undefined.');
514                 retVal = null;
515             }
516             return retVal;
517         }
518 
519         var value = mixed_val;
520 
521         var quote = function (string) {
522             var escapable = /[\\\"\u0000-\u001f\u007f-\u009f\u00ad\u0600-\u0604\u070f\u17b4\u17b5\u200c-\u200f\u2028-\u202f\u2060-\u206f\ufeff\ufff0-\uffff]/g;
523             var meta = { // table of character substitutions
524                 '\b': '\\b',
525                 '\t': '\\t',
526                 '\n': '\\n',
527                 '\f': '\\f',
528                 '\r': '\\r',
529                 '"': '\\"',
530                 '\\': '\\\\'
531             };
532 
533             escapable.lastIndex = 0;
534             return escapable.test(string) ? '"' + string.replace(escapable, function (a) {
535                 var c = meta[a];
536                 return typeof c === 'string' ? c : '\\u' + ('0000' + a.charCodeAt(0).toString(16)).slice(-4);
537             }) + '"' : '"' + string + '"';
538         };
539 
540         var str = function (key, holder) {
541             var gap = '';
542             var indent = '    ';
543             var i = 0; // The loop counter.
544             var k = ''; // The member key.
545             var v = ''; // The member value.
546             var length = 0;
547             var mind = gap;
548             var partial = [];
549             var value = holder[key];
550 
551             // If the value has a toJSON method, call it to obtain a replacement value.
552             if (value && typeof value === 'object' && typeof value.toJSON === 'function') {
553                 value = value.toJSON(key);
554             }
555 
556             // What happens next depends on the value's type.
557             switch (typeof value) {
558             case 'string':
559                 return quote(value);
560 
561             case 'number':
562                 // JSON numbers must be finite. Encode non-finite numbers as null.
563                 return isFinite(value) ? String(value) : 'null';
564 
565             case 'boolean':
566             case 'null':
567                 // If the value is a boolean or null, convert it to a string. Note:
568                 // typeof null does not produce 'null'. The case is included here in
569                 // the remote chance that this gets fixed someday.
570                 return String(value);
571 
572             case 'object':
573                 // If the type is 'object', we might be dealing with an object or an array or
574                 // null.
575                 // Due to a specification blunder in ECMAScript, typeof null is 'object',
576                 // so watch out for that case.
577                 if (!value) {
578                     return 'null';
579                 }
580 //                 if ((this.PHPJS_Resource && value instanceof this.PHPJS_Resource) || (window.PHPJS_Resource && value instanceof window.PHPJS_Resource)) {
581 //                     throw new SyntaxError('json_encode');
582 //                 }
583 
584                 // Make an array to hold the partial results of stringifying this object value.
585                 gap += indent;
586                 partial = [];
587 
588                 // Is the value an array?
589                 if (Object.prototype.toString.apply(value) === '[object Array]') {
590                     // The value is an array. Stringify every element. Use null as a placeholder
591                     // for non-JSON values.
592                     length = value.length;
593                     for (i = 0; i < length; i += 1) {
594                         partial[i] = str(i, value) || 'null';
595                     }
596 
597                     // Join all of the elements together, separated with commas, and wrap them in
598                     // brackets.
599                     v = partial.length === 0 ? '[]' : gap ? '[\n' + gap + partial.join(',\n' + gap) + '\n' + mind + ']' : '[' + partial.join(',') + ']';
600                     gap = mind;
601                     return v;
602                 }
603 
604                 // Iterate through all of the keys in the object.
605                 for (k in value) {
606                     if (Object.hasOwnProperty.call(value, k)) {
607                         v = str(k, value);
608                         if (v) {
609                             partial.push(quote(k) + (gap ? ': ' : ':') + v);
610                         }
611                     }
612                 }
613 
614                 // Join all of the member texts together, separated with commas,
615                 // and wrap them in braces.
616                 v = partial.length === 0 ? '{}' : gap ? '{\n' + gap + partial.join(',\n' + gap) + '\n' + mind + '}' : '{' + partial.join(',') + '}';
617                 gap = mind;
618                 return v;
619             case 'undefined':
620                 // Fall-through
621             case 'function':
622                 /* falls through */
623             default:
624                 fCore.debug('Cannot encode type %s', typeof value);
625             }
626         };
627 
628         // Make a fake root object containing our value under the key of ''.
629         // Return the result of stringifying the value.
630         return str('', {
631             '': value
632         });
633 
634     } catch (err) {
635         // (i.e., when the JSON global is not available and there is an error)
636         if (!(err instanceof SyntaxError)) {
637             fCore.debug('Unexpected error type in non-native JSON encoder. Returning null.');
638         }
639         fJSON.lastError = 4;
640     }
641     return '';
642 };
