 77 /**
 78  * Checks a string of HTML for block level elements. Elements that are by
 79  *   default not counted as block-level:
 80  * <ul>
 81  * <li><code><link></code></li>
 82  * <li><code><nav></code></li>
 83  * <li><code><optgroup></code></li>
 84  * <li><code><option></code></li>
 85  * <li><code><summary></code></li>
 86  * <li><code><tbody></code></li>
 87  * <li><code><td></code></li>
 88  * </ul>
 89  * To check for these elements, set the <code>strict</code> argument to
 90  *   <code>true</code>.
 91  * @param {string} content The HTML content to check.
 92  * @param {boolean} [strict=false] Do not allow exceptions.
 93  * @returns {boolean} If the content has a block level tag.
 94  */
 95 fHTML.containsBlockLevelHTML = function (content, strict) {
 96   if (strict === undefined) {
 97     strict = false;
 98   }
 99 
100   // NOTE This checks both HTML 5 and previous versions
101   var inline = fHTML._inlineElements;
102 
103   // Many might consider these block-level elements in practise
104   if (strict) {
105     inline = inline.concat.apply(inline, fHTML._inlineElementsStrict);
106   }
107 
108   var container = document.createElement('div');
109   container.innerHTML = content;
110 
111   return (function check(nodes) {
112     for (var i = 0; i < nodes.length; i++) {
113       if (nodes[i].nodeName.substr(0, 1) === '#') {
114         continue;
115       }
116 
117       if (!inArray(nodes[i].nodeName, inline)) {
118         return true;
119       }
120 
121       if (nodes[i].hasChildNodes()) {
122         return check(nodes[i].childNodes);
123       }
124     }
125 
126     return false;
127   })(container.childNodes);
128 };
129 /**
130  * Converts new lines to <code><br></code> tags as long as there are not
131  *   any block-level HTML tags in the content.
132  * @param {string} content The content to display.
133  * @returns {string} The content.
134  */
135 fHTML.convertNewLines = function (content) {
136   if (fHTML.containsBlockLevelHTML(content)) {
137     return content;
138   }
139 
140   var offset = arrayIndexOf('br', fHTML._inlineElements);
141   var inlineWithoutBr = fHTML._inlineElements.slice(0, offset);
142   var add = fHTML._inlineElements.slice(offset + 1);
143   inlineWithoutBr = inlineWithoutBr.concat.apply(inlineWithoutBr, add);
144 
145   // http://phpjs.org/functions/nl2br:480
146   return content.replace(/([^>\r\n]?)(\r\n|\n\r|\r|\n)/g, '$1' + '<br>' + '$2');
147 };

269 /**
270  * Converts all HTML entities to normal characters.
271  * @param {string} content The content to decode.
272  * @returns {string} The decoded content.
273  */
274 fHTML.decode = function (content) {
275   var entity, tmpStr = content.toString();
276 
277   for (var symbol in fHTML._decodeHashMap) {
278     if (fHTML._decodeHashMap.hasOwnProperty(symbol)) {
279       entity = fHTML._decodeHashMap[symbol];
280       tmpStr = tmpStr.split(entity).join(symbol);
281     }
282   }
283   tmpStr = tmpStr.split(''').join("'");
284 
285   return tmpStr;
286 };
287 /**
288  * Converts all special characters to entities.
289  * @param {string|Array} content The content to encode.
290  * @returns {string} The encoded content.
291  */
292 fHTML.encode = function (content) {
293   if (typeof content === 'string') {
294     content = [content];
295   }
296 
297   // Mini version of htmlspecialchars()
298   var specialChars = function (str) {
299     var ret = str.replace(/</g, '<').replace(/>/g, '>');
300     ret = ret.replace(/"/g, '"');
301     ret = ret.replace(/'/g, ''');
302     return ret;
303   };
304   for (var i = 0; i < content.length; i++) {
305     content[i] = specialChars(content[i].toString());
306   }
307 
308   return content.join('');
309 };
310 /**
311  * Make links within a content string.
312  * @param {string} content Content to process.
313  * @param {number} [linkTextLength] Length of the text within the tag. If this
314  *   argument is used, then a title attribute will be on each link with the
315  *   full link text.
316  * @returns {string}
317  */
318 fHTML.makeLinks = function(content, linkTextLength) {
319   var div = document.createElement('div');
320   var regex = /\b([a-z]{3,}:\/\/[a-z0-9%\$\-_.+!*;\/?:@=&\'\#,]+[a-z0-9\$\-_+!*;\/?:@=&\'\#,])\b|\b(www\.(?:[a-z0-9\-]+\.)+[a-z]{2,}(?:\/[a-z0-9%\$\-_.+!*;\/?:@=&\'\#,]+[a-z0-9\$\-_+!*;\/?:@=&\'\#,])?)\b|\b([a-z0-9\\.+\'_\\\-]+@(?:[a-z0-9\\\-]+\.)+[a-z]{2,})\b/gi;
321   div.innerHTML = content; // Encodes everything
322 
323   content = div.innerHTML;
324   return (function make(nodes) {
325     var currentText = '', tmp;
326     var matches;
327     var printedText, scheme, replace;
328     var container, tag;
329     var matchRegex;
330     var used = {};
331 
332     // Find every #text node and parse URIs
333     for (var i = 0, j; i < nodes.length; i++) {
334       if (nodes[i].hasChildNodes()) {
335         // Make a new tag to get innerHTML from
336         // This preserves outer tags without having to rely on outerHTML being available
337         container = document.createElement('div');
338         tag = nodes[i].cloneNode(true);
339         tag.innerHTML = make(nodes[i].childNodes);
340         container.appendChild(tag);
341         currentText += container.innerHTML;
342       }
343       else if (nodes[i].nodeType === 3) {
344         currentText += nodes[i].nodeValue;
345 
346         matches = currentText.match(regex);
347         if (matches !== null && matches.length) {
348           for (j = 0; j < matches.length; j++) {
349             if (used[matches[j]] !== undefined) {
350               continue;
351             }
352 
353             printedText = matches[j];
354 
355             if (linkTextLength !== undefined) {
356               printedText = matches[j].substr(0, linkTextLength) + '...';
357             }
358 
359             // Simple check
360             scheme = '';
361             if (matches[j].indexOf('@') !== -1) {
362               scheme = 'mailto:';
363             }
364 
365             replace = '<a href="' + scheme + matches[j] + '" ';
366             if (linkTextLength !== undefined) {
367               replace += 'title="' + matches[j] + '"';
368             }
369             replace += '>' + printedText + '</a>';
370 
371             matchRegex = new RegExp(matches[j], 'g');
372             used[matches[j]] = matches[j];
373             currentText = currentText.replace(matchRegex, replace);
374           }
375         }
376       }
377     }
378 
379     return currentText;
380   })(div.childNodes);
381 };
382 /**

384  * @param {string|Array} content The content to prepare.

387 fHTML.prepare = function (content) {
388   if (typeof content !== 'object') {
