  7 var fCryptography = function () {};
  8 /**
  9  * Generates a random number. If either parameter passed is invalid
 10  *   (<code>NaN</code>), then <code>Math.random().</code> will be called.
 11  * @param {number} [min=0] The minimum number to generate.
 12  * @param {number} [max=1] The maximum number to generate.
 13  * @returns {number} A random number.
 14  */
 15 fCryptography.random = function (min, max) {
 16   min = parseInt(min, 10);
 17   max = parseInt(max, 10);
 18 
 19   if(!isNaN(min) && !isNaN(max)) {
 20     return Math.floor(Math.random() * (max - min + 1)) + min;
 21   }
 22 
 23   return Math.random();
 24 };
 25 /**
 26  * Generates a random string.
 27  * @param {number} [length=32] The length you want the string to be.
 28  * @param {string} [type='base64'] The alphabet of string you want generated.
 29  *   One of: <code>'alphanumeric'</code>, <code>'base56'</code>,
 30  *   <code>'alpha'</code>, <code>'base36'</code>, <code>'hexadecimal'</code>,
 31  *   <code>'numeric'</code>, <code>'base64'</code>.
 32  * @returns {string} Random string.
 33 */
 34 fCryptography.randomString = function (length, type) {
 35   if (!length) {
 36     length = 32;
 37   }
 38 
 39   var digits = '0123456789';
 40   var abcdef = 'abcdef';
 41   var upperAlpha = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ';
 42   var alphabet = upperAlpha.toLowerCase() + upperAlpha + digits + '+/';
 43   var output = '';
 44 
 45   if (type !== undefined) {
 46     if (type === 'alphanumeric') {
 47       alphabet = upperAlpha.toLowerCase() + upperAlpha + digits;
 48     }
 49     else if (type === 'base56') {
 50       alphabet = upperAlpha.toLowerCase() + upperAlpha + digits.substr(2);
 51     }
 52     else if (type === 'alpha') {
 53       alphabet = upperAlpha.toLowerCase() + upperAlpha;
 54     }
 55     else if (type === 'base36') {
 56       alphabet = upperAlpha + digits;
 57     }
 58     else if (type === 'hexadecimal') {
 59       alphabet = abcdef + digits;
 60     }
 61     else if (type === 'numeric') {
 62       alphabet = digits;
 63     }
 64   }
 65 
 66   for (var i = 0, alphaLen = alphabet.length - 1; i < length; i++) {
 67     output += alphabet.charAt(fCryptography.random(0, alphaLen));
 68   }
 69 
 70   return output;
 71 };
